Lab9 DataTypes&Variable: ระบุค่า
ให้ระบุค่าของ a, b, c , d และ e หลังจากจบการทำงานในแต่ละ statements


let a = 1;                   // a=1
let b = 2;                   // b=2
let c = a++;                 // c=2
let d = ++c;                 // d=2
let e = ++d + d++ + d;       // e=10
